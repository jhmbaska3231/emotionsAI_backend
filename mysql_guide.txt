SELECT 
    u.user_id,
    u.user_type,
    d.diary_id,
    d.date,
    d.input_text,
    d.emotional_intensity,
    d.overall_sentiment,
    te.emotion,
    te.emotion_percentage
FROM 
    diary d
LEFT JOIN 
    target_emotion te ON d.diary_id = te.diary_id
LEFT JOIN 
    user u ON d.user_id = u.user_id;
